Option Explicit

Private Sub CommandButton1_Click()
    ActiveCell.Value = "Price"
    ActiveCell.Offset(0, 1).Value = tbPrice.Value
    ActiveCell.Offset(1, 0).Value = "Discount (%)"
    ActiveCell.Offset(1, 1).Value = tbPerDiscount.Value
    ActiveCell.Offset(2, 0).Value = "Net Price"
    ActiveCell.Offset(2, 1).Formula2R1C1 = "=R[-2]C * (1 - (R[-1]C / 100))"
    
    Me.Hide
End Sub
